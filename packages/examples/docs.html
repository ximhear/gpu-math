gpu-math

WebGPU-powered interactive math visualization

Examples Documentation

Quick Start

Installation

npm install gpu-math

Your First Plot

import { createScene, plot } from 'gpu-math';

const canvas = document.getElementById('canvas');
const scene = await createScene(canvas);
scene.add(plot(x => Math.sin(x)));

That's it! You get a smooth sine wave on an infinite grid. Drag to pan, scroll to zoom.

Scene

2D Scene

const scene = await createScene(canvas, {
  width: 800,          // canvas width in CSS pixels
  height: 600,         // canvas height in CSS pixels
  theme: '3b1b',       // '3b1b' | 'light' | 'dark' | 'minimal' | custom Theme object
  interactive: true,   // enable pan/zoom (default: true)
  pixelRatio: 'auto',  // 'auto' | number (default: 'auto')
});

scene.add(someObject);    // add a math object
scene.remove(someObject); // remove it
scene.destroy();          // clean up GPU resources

3D Scene

import { createScene3D, surface } from 'gpu-math';

const scene3d = await createScene3D(canvas, {
  width: 800,
  height: 600,
  fov: 45,  // field of view in degrees
});

// Drag to rotate, scroll to zoom
scene3d.add(surface((u, v) => [u, Math.sin(u) * Math.cos(v), v], {
  u: [-3, 3],
  v: [-3, 3],
}));

2D Objects

plot(fn, options?)

Plot a function y = f(x).

// Simple
scene.add(plot(x => Math.sin(x)));

// With options
scene.add(plot(x => x ** 2, {
  range: [-5, 5],     // x range (default: [-10, 10])
  color: '#3b82f6',   // line color
  lineWidth: 2,       // line thickness in pixels
  dash: [8, 4],       // [dash length, gap length] — solid if omitted
  opacity: 1,         // 0 to 1
  label: 'x²',        // legend label
  samples: 512,       // number of initial samples
}));

// With parameters (for animation)
const p = plot((x, { a, b }) => a * Math.sin(b * x), {
  params: { a: 1, b: 1 },
  label: 'a·sin(b·x)',
});
scene.add(p);

// Later: change parameters
p.setParam('a', 2);
p.setParam('b', 3);

parametric(fn, options?)

Plot a parametric curve t → [x, y].

// Circle
scene.add(parametric(t => [Math.cos(t), Math.sin(t)], {
  t: [0, 2 * Math.PI],  // parameter range
  color: '#10b981',
  lineWidth: 2,
  label: 'circle',
  samples: 256,
}));

// Lissajous curve
scene.add(parametric(t => [Math.sin(3*t), Math.sin(2*t)], {
  t: [0, 2 * Math.PI],
}));

// Spiral
scene.add(parametric(t => {
  const r = 0.2 * t;
  return [r * Math.cos(t), r * Math.sin(t)];
}, { t: [0, 6 * Math.PI] }));

point(position, options?)

scene.add(point([1, 0], {
  color: '#ef4444',
  size: 8,         // radius in pixels
  label: 'A',
}));

vector(from, to, options?)

scene.add(vector([0, 0], [2, 1], {
  color: '#10b981',
  lineWidth: 2,
  headSize: 0.15,  // arrowhead size in world units
  label: 'v',
}));

line(from, to, options?)

scene.add(line([0, 0], [3, 2], {
  color: '#6b7280',
  lineWidth: 1,
  dash: [6, 4],
}));

vectorField(fn, options?)

Render a 2D vector field as a grid of arrows.

// Rotation field
scene.add(vectorField((x, y) => [-y, x], {
  density: 20,           // arrows per axis (default: 15)
  scale: 0.5,            // arrow length scale (default: 0.8)
  color: '#8b5cf6',
  lineWidth: 1.5,
  opacity: 0.7,
}));

// Source field
scene.add(vectorField((x, y) => [x, y], {
  density: 14,
  color: '#ef4444',
}));

3D Objects

surface(fn, options?)

Render a parametric surface (u, v) → [x, y, z].

scene3d.add(surface((u, v) => [u, Math.sin(u) * Math.cos(v), v], {
  u: [-3, 3],          // u parameter range
  v: [-3, 3],          // v parameter range
  resolution: 64,      // grid resolution (default: 64)
  wireframe: true,     // show wireframe overlay (default: false)
  opacity: 1,          // 0 to 1
}));

// Paraboloid
scene3d.add(surface((u, v) => [u, u*u + v*v, v], {
  u: [-2, 2], v: [-2, 2],
}));

Linear Algebra

transform(matrix, options?)

Visualize a 2×2 matrix transformation on the grid.

import { transform, animateParam } from 'gpu-math';

// Shear matrix
const t = transform([[2, 1], [0, 1]], {
  showGrid: true,      // show transformed grid lines
  showEigen: true,     // show eigenvectors
  color: '#3b82f6',    // grid color
  gridRange: 5,        // grid extent
  gridDensity: 11,     // number of grid lines
});
scene.add(t);

// Animate: identity → full transform
t.setProgress(0);
await animateParam(t, 't', {
  from: 0, to: 1, duration: 2000, easing: 'easeInOut',
});

Calculus

tangentLine(plot, options)

Draw the tangent line at a specific point on a plot.

const f = plot(x => Math.sin(x), { lineWidth: 2 });
scene.add(f);

scene.add(tangentLine(f, {
  at: 1,             // x-coordinate of tangent point
  color: '#ef4444',
  lineWidth: 2,
  length: 4,         // tangent line length in world units
}));

// Animate the tangent point
const tl = tangentLine(f, { at: 0 });
scene.add(tl);
await animateParam(tl, 'at', { from: -3, to: 3, duration: 3000 });

riemannSum(plot, options)

Visualize Riemann sums (rectangles approximating area under a curve).

const f = plot(x => x * x * 0.3, { label: '0.3x²' });
scene.add(f);

const rs = riemannSum(f, {
  from: 0,           // left bound
  to: 3,             // right bound
  n: 10,             // number of rectangles
  method: 'left',    // 'left' | 'right' | 'midpoint'
  color: '#10b981',
  opacity: 0.4,
});
scene.add(rs);

// Animate: increase rectangles to show convergence
await animateParam(rs, 'n', { from: 5, to: 100, duration: 3000 });

Animation

animate(options)

Morph a plot from its current function to a new one.

const p = plot(x => Math.sin(x));
scene.add(p);

await animate({
  from: p,
  to: x => Math.sin(3 * x),  // target function
  duration: 1000,              // milliseconds
  easing: 'easeInOut',         // easing function
});

animateParam(target, param, options)

Animate a numeric parameter over time.

const p = plot((x, { freq }) => Math.sin(freq * x), { params: { freq: 1 } });
scene.add(p);

await animateParam(p, 'freq', {
  from: 1,          // start value (default: current value)
  to: 5,            // end value
  duration: 2000,
  easing: 'easeInOut',
});

sequence(steps) + wait(ms)

Run animations in order.

await sequence([
  () => animateParam(p, 'freq', { to: 5, duration: 1000 }),
  () => wait(500),
  () => animate({ from: p, to: x => Math.cos(x), duration: 800 }),
  () => wait(300),
  () => animateParam(p, 'freq', { to: 1, duration: 1000, easing: 'bounceOut' }),
]);

Easing Functions

Available easing names:

NameDescription
linearConstant speed
easeInSlow start (cubic)
easeOutSlow end (cubic)
easeInOutSlow start and end (cubic)
bounceOutBounce effect at the end
elasticOutSpring/elastic effect at the end

You can also pass a custom easing function (t: number) => number.

React

Installation

npm install gpu-math @gpu-math/react

Usage

import { MathCanvas, Plot, Surface, Point, VectorField, Parametric } from '@gpu-math/react';

// 2D
function App() {
  return (
    <MathCanvas width={800} height={600} theme="3b1b">
      <Plot fn={x => Math.sin(x)} color="blue" label="sin(x)" />
      <Plot fn={x => Math.cos(x)} color="red" label="cos(x)" />
      <Point at={[0, 0]} color="white" size={5} />
    </MathCanvas>
  );
}

// 3D
function App3D() {
  return (
    <MathCanvas width={800} height={600} dimension={3}>
      <Surface fn={(u, v) => [u, Math.sin(u) * Math.cos(v), v]}
        u={[-3, 3]} v={[-3, 3]} wireframe />
    </MathCanvas>
  );
}

// Interactive parameters with React state
function Wave() {
  const [freq, setFreq] = useState(1);
  return (
    <>
      <input type="range" min={1} max={10} value={freq}
        onChange={e => setFreq(+e.target.value)} />
      <MathCanvas width={600} height={400}>
        <Plot fn={x => Math.sin(freq * x)} color="blue" />
      </MathCanvas>
    </>
  );
}
